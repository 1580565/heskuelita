-- Table: public.professor

-- DROP TABLE public.professor;

CREATE TABLE public.professor
(
    id_prof integer NOT NULL DEFAULT nextval('professor_seq'::regclass),
    first_name character varying(20) COLLATE pg_catalog."default" NOT NULL,
    last_name character varying(20) COLLATE pg_catalog."default" NOT NULL,
    address character varying(250) COLLATE pg_catalog."default" NOT NULL,
    gender character(1) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT professor_pkey PRIMARY KEY (id_prof)
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public.professor
    OWNER to postgres;
